script.js:
document.getElementById('certificadoInput').addEventListener('change', handleFileSelect);


let files = [];
let currentPDFIndex = 0;
let currentSectionIndex = 0;
let pdfDoc = null;

function handleFileSelect(event) {
    files = Array.from(event.target.files);
    const thumbnails = document.getElementById('thumbnails');
    thumbnails.innerHTML = '';

    files.forEach((file, index) => {
        const reader = new FileReader();
        reader.onload = function() {
            const pdfData = new Uint8Array(reader.result);
            generateThumbnail(pdfData, index);
        };
        reader.readAsArrayBuffer(file);
    });
}

function generateThumbnail(pdfData, index) {
    const loadingTask = pdfjsLib.getDocument({ data: pdfData });
    loadingTask.promise.then(function(pdf) {
        pdf.getPage(1).then(function(page) {
            const scale = 0.3; 
            const viewport = page.getViewport({ scale: scale });
            const canvas = document.createElement('canvas');
            const context = canvas.getContext('2d');
            canvas.height = viewport.height;
            canvas.width = viewport.width;

            const renderContext = {
                canvasContext: context,
                viewport: viewport
            };
            page.render(renderContext).promise.then(function() {
                const img = document.createElement('img');
                img.src = canvas.toDataURL();
                img.alt = `PDF ${index + 1}`;
                img.setAttribute('data-index', index);
                img.style.cursor = 'pointer';
                img.addEventListener('click', () => loadPDF(index));
                document.getElementById('thumbnails').appendChild(img);
            }).catch(function(error) {
                console.error('Erro ao renderizar a página:', error);
            });
        }).catch(function(error) {
            console.error('Erro ao obter a página do PDF:', error);
        });
    }).catch(function(error) {
        console.error('Erro ao carregar o documento PDF:', error);
    });
}

function loadPDF(index) {
    currentPDFIndex = index;
    const file = files[index];
    const reader = new FileReader();
    reader.onload = function() {
        const pdfData = new Uint8Array(reader.result);
        pdfjsLib.getDocument({ data: pdfData }).promise.then(function(pdf) {
            pdfDoc = pdf;
            currentSectionIndex = 0; 
            displaySection(currentSectionIndex);
            extractAndDisplayData(pdfData); // Extraia e exiba os dados
        }).catch(function(error) {
            console.error('Erro ao carregar o documento PDF:', error);
        });
    };
    reader.readAsArrayBuffer(file);
}
